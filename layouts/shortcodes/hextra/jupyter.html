{{- /* Hextra Jupyter Notebook shortcode (Bootstrap 5 port). Renders a Jupyter notebook (.ipynb) as code blocks and markdown. @param {string} 0 Path or URL of the Jupyter notebook. @param {bool} allowUnsafeHTML Allow raw HTML from notebook outputs (default: false). @example {{% hextra/jupyter "notebook.ipynb" %}} */ -}} {{- $path := .Get 0 -}} {{- $allowUnsafeHTML := eq (.Get "allowUnsafeHTML") "true" -}} {{- $filename := .Get "filename" -}} {{- $fileAttr := "" -}} {{- if $filename -}}{{- $fileAttr = printf " {filename=\"%s\"}" $filename -}}{{- end -}} {{- $data := "" -}} {{- $page := .Page -}} {{- $isLocal := not (urls.Parse $path).Scheme -}} {{- $isPage := and (eq .Page.Kind "page") (not .Page.BundleType) -}} {{- if (not $isLocal) -}} {{- with resources.GetRemote $path -}} {{- with unmarshal .Content -}}{{- $data = . -}}{{- end -}} {{- else -}} {{- errorf "Remote resource not found: %s" $path -}} {{- end -}} {{- else if (not $isPage) -}} {{- with .Page.Resources.Get $path -}} {{- with unmarshal .Content -}}{{- $data = . -}}{{- end -}} {{- else -}} {{- errorf "Local resource not found: %s" $path -}} {{- end -}} {{- else -}} {{- with resources.Get $path -}} {{- with unmarshal .Content -}}{{- $data = . -}}{{- end -}} {{- else -}} {{- errorf "Local resource not found: %s" $path -}} {{- end -}} {{- end -}} {{- $language := index $data "metadata" "language_info" "name" | default "python" -}} {{- with index $data "cells" -}} {{- range $cell := . -}} {{- if eq (index $cell "cell_type") "code" -}} {{- $source := index $cell "source" -}} {{- $sourceContent := (cond (reflect.IsSlice $source) (delimit $source "") $source) -}} {{- with ($sourceContent | strings.Chomp) -}} {{ (printf "\n\n```%s%s\n%s\n```\n" $language $fileAttr .) | safeHTML -}} {{- end -}}
{{- $outputs := index $cell "outputs" -}} {{- with $outputs -}}
{{- range $output := . -}} {{- if eq (index $output "output_type") "display_data" -}} {{- $imgData := index $output "data" -}} {{- $image := index $imgData "image/png" -}} {{- if $image -}} {{- end -}} {{- else if eq (index $output "output_type") "stream" -}} {{- $text := index $output "text" -}} {{- $textContent := (cond (reflect.IsSlice $text) (delimit $text "") $text) -}} 
{{- $textContent -}}
{{- else if eq (index $output "output_type") "execute_result" -}} {{- $resultData := index $output "data" -}} {{- $text := index $resultData "text/plain" -}} {{- $textContent := (cond (reflect.IsSlice $text) (delimit $text "") $text) -}} 
{{- $textContent -}}
{{- if $allowUnsafeHTML -}} {{- $html := index $resultData "text/html" -}} {{- if $html -}} {{- $htmlText := delimit $html "" -}}
{{- $htmlText | safeHTML -}}
{{- end -}} {{- end -}} {{- end -}} {{- end -}}
{{- end -}}
{{- else if eq (index $cell "cell_type") "markdown" -}} {{- $source := index $cell "source" -}} {{- $sourceContent := (cond (reflect.IsSlice $source) (delimit $source "") $source) -}} {{ $page.RenderString $sourceContent }} {{- end -}} {{- end -}} {{- end -}}
